cmake_minimum_required(VERSION 3.10)

set(BLAS_INCLUDE_DIRS
        .
)

set(BLAS_HEADERS
        blas.h
        axpy.h
        gemm.h
        gemm_1c.h
        gemm_oc.h
)

)

set(BLAS_SOURCES ${BLAS_SOURCES} PARENT_SCOPE)
